import React from 'react';
import { Plus, Trash2, GraduationCap } from 'lucide-react';

/**
 * STEP 4: EDUCATION COMPONENT
 * Manages multiple education entries
 * 
 * Props:
 * - education: Array of education objects
 * - setEducation: Function to update education array
 */

export default function EducationStep({ education, setEducation }) {
  
  // Add new education entry
  const addEducation = () => {
    setEducation([...education, { 
      id: Date.now(), 
      school: '', 
      degree: '', 
      field: '', 
      startDate: '', 
      endDate: '', 
      gpa: '' 
    }]);
  };

  // Remove education entry
  const removeEducation = (id) => {
    if (education.length > 1) {
      setEducation(education.filter(edu => edu.id !== id));
    }
  };

  // Update specific field in education entry
  const updateEducation = (id, field, value) => {
    setEducation(education.map(edu => 
      edu.id === id ? { ...edu, [field]: value } : edu
    ));
  };

  return (
    <div className="space-y-5 animate-fadeIn">
      {/* Header with Add Button */}
      <div className="flex justify-between items-center">
        <p className="text-sm text-gray-600">
          Add your educational background, starting with the most recent.
        </p>
        <button
          onClick={addEducation}
          className="flex items-center gap-2 bg-gradient-to-r from-indigo-600 to-purple-600 text-white px-4 py-2 rounded-lg hover:from-indigo-700 hover:to-purple-700 transition-all duration-200 hover:scale-105 shadow-md text-sm"
        >
          <Plus size={16} />
          Add
        </button>
      </div>

      {/* Education Entries */}
      {education.map((edu, index) => (
        <div 
          key={edu.id} 
          className="bg-gradient-to-br from-gray-50 to-indigo-50 p-5 rounded-xl space-y-4 border-2 border-gray-200 hover:border-indigo-300 transition-all"
        >
          {/* Entry Header */}
          <div className="flex justify-between items-center">
            <h4 className="font-semibold text-gray-800 flex items-center gap-2">
              <GraduationCap size={18} className="text-indigo-600" />
              Education {index + 1}
            </h4>
            {education.length > 1 && (
              <button
                onClick={() => removeEducation(edu.id)}
                className="text-red-500 hover:text-red-700 hover:bg-red-50 p-2 rounded-lg transition-all"
              >
                <Trash2 size={18} />
              </button>
            )}
          </div>

          {/* School/University */}
          <div>
            <label className="block text-xs font-semibold text-gray-600 mb-1">
              School/University Name <span className="text-red-500">*</span>
            </label>
            <input
              type="text"
              placeholder="Harvard University, MIT, Stanford, etc."
              value={edu.school}
              onChange={(e) => updateEducation(edu.id, 'school', e.target.value)}
              className="w-full px-3 py-2 border-2 border-gray-200 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition-all"
            />
          </div>

          {/* Degree and Field of Study */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div>
              <label className="block text-xs font-semibold text-gray-600 mb-1">
                Degree <span className="text-red-500">*</span>
              </label>
              <input
                type="text"
                placeholder="Bachelor's, Master's, Ph.D."
                value={edu.degree}
                onChange={(e) => updateEducation(edu.id, 'degree', e.target.value)}
                className="w-full px-3 py-2 border-2 border-gray-200 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition-all"
              />
            </div>
            <div>
              <label className="block text-xs font-semibold text-gray-600 mb-1">
                Field of Study
              </label>
              <input
                type="text"
                placeholder="Computer Science, Business, etc."
                value={edu.field}
                onChange={(e) => updateEducation(edu.id, 'field', e.target.value)}
                className="w-full px-3 py-2 border-2 border-gray-200 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition-all"
              />
            </div>
          </div>

          {/* Dates and GPA */}

              </label>
              <input

/**
 * STEP 5: SKILLS COMPONENT
 * Manages three categories of skills: Technical, Languages, and Soft Skills
 * 
 * Props:
 * - skills: Object with three arrays (technical, languages, soft)
 * - setSkills: Function to update skills object
 */

export default function SkillsStep({ skills, setSkills }) {
  
  // Add skill to specific category
  const addSkill = (category) => {
    const categoryNames = {
      technical: 'Technical Skill',
      languages: 'Language',
      soft: 'Soft Skill'
    };
    
    const skill = prompt(`Add ${categoryNames[category]}:`);
    if (skill && skill.trim()) {
      setSkills({
        ...skills,
        [category]: [...skills[category], skill.trim()]
      });
    }
  };

  // Remove skill from specific category
  const removeSkill = (category, index) => {
    setSkills({
      ...skills,
      [category]: skills[category].filter((_, i) => i !== index)
    });
  };

  return (
    <div className="space-y-6 animate-fadeIn">
      {/* Introduction */}
      <p className="text-sm text-gray-600">
        Add your skills in different categories. Click "Add" to input each skill one at a time.
      </p>

      {/* TECHNICAL SKILLS */}
      <div className="bg-gradient-to-br from-blue-50 to-cyan-50 p-5 rounded-xl border-2 border-blue-200 hover:border-blue-400 transition-all">
        <div className="flex justify-between items-center mb-4">
          <div>
            <h4 className="font-semibold text-blue-900 flex items-center gap-2 text-base">
              üíª Technical Skills
            </h4>
            <p className="text-xs text-blue-700 mt-1">
              Programming languages, frameworks, tools, software
            </p>
          </div>
          <button
            onClick={() => addSkill('technical')}
            className="flex items-center gap-1 bg-blue-600 text-white px-3 py-1.5 rounded-lg text-sm hover:bg-blue-700 transition-all hover:scale-105 shadow-md"
          >
            <Plus size={16} />
            Add
          </button>
        </div>
        
        <div className="flex flex-wrap gap-2">
          {skills.technical.length === 0 ? (
            <p className="text-sm text-blue-700 italic">
              Click "Add" to add technical skills
            </p>
          ) : (
            skills.technical.map((skill, index) => (
              <span
                key={index}
                className="bg-blue-600 text-white px-3 py-1.5 rounded-full text-sm flex items-center gap-2 shadow-sm hover:shadow-md transition-all"
              >
                {skill}
                <button
                  onClick={() => removeSkill('technical', index)}
                  className="hover:bg-blue-700 rounded-full w-5 h-5 flex items-center justify-center"
                  title="Remove skill"
                >
                  √ó
                </button>
              </span>
            ))
          )}
        </div>
        
        {/* Examples */}
        <div className="mt-3 bg-blue-100 p-2 rounded text-xs text-blue-800">
          <strong>Examples:</strong> JavaScript, Python, React, SQL, AWS, Docker, Git
        </div>
      </div>

      {/* LANGUAGES */}
      <div className="bg-gradient-to-br from-green-50 to-emerald-50 p-5 rounded-xl border-2 border-green-200 hover:border-green-400 transition-all">
        <div className="flex justify-between items-center mb-4">
          <div>
            <h4 className="font-semibold text-green-900 flex items-center gap-2 text-base">
              üåç Languages
            </h4>
            <p className="text-xs text-green-700 mt-1">
              Spoken and written languages with proficiency levels
            </p>
          </div>
          <button
            onClick={() => addSkill('languages')}
            className="flex items-center gap-1 bg-green-600 text-white px-3 py-1.5 rounded-lg text-sm hover:bg-green-700 transition-all hover:scale-105 shadow-md"

        </div>

{currentStep === 5 && (
  <ProjectsStep 
    projects={projects}
    setProjects={setProjects}
  />
)}

{currentStep === 6 && (
  <CertificationsStep 
    certifications={certifications}
    setCertifications={setCertifications}
  />
)}

{currentStep === 7 && (
  <AdditionalSectionsStep 
    customSections={customSections}
    setCustomSections={setCustomSections}
  />
)}

{currentStep === 8 && (
  <ReviewDownloadStep 
    personalInfo={personalInfo}
    experience={experience}
    education={education}
    skills={skills}
    projects={projects}
    certifications={certifications}
    customSections={customSections}
    downloadResume={downloadResume}
    isStepComplete={isStepComplete}
  />
)}



Helper Functions
// Navigation
const nextStep = () => {
  if (currentStep < 8) {
    setCurrentStep(currentStep + 1);
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }
};

const prevStep = () => {
  if (currentStep > 0) {
    setCurrentStep(currentStep - 1);
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }
};

// Validation
const isStepComplete = (stepId) => {
  switch(stepId) {
    case 0: return personalInfo.fullName && personalInfo.email;
    case 1: return personalInfo.summary.length > 20;
    case 2: return experience.some(exp => exp.company && exp.position);
    case 3: return education.some(edu => edu.school && edu.degree);
    case 4: return skills.technical.length > 0 || skills.languages.length > 0 || skills.soft.length > 0;
    case 5: return projects.some(proj => proj.name);
    case 6: return certifications.some(cert => cert.name);
    case 7: return true;
    default: return false;
  }
};

// Download
const downloadResume = () => {
  const resumeData = {
    personalInfo,
    experience,
    education,
    skills,
    projects,
    certifications,
    customSections,
    generatedDate: new Date().toISOString()
  };
  
  const blob = new Blob([JSON.stringify(resumeData, null, 2)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `resume-${personalInfo.fullName.replace(/\s+/g, '-').toLowerCase() || 'data'}.json`;
  a.click();
  URL.revokeObjectURL(url);
};
